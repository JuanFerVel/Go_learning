package main

import (
	"database/sql"
	"log"

	_ "github.com/go-sql-driver/mysql" // la "_" es porque se esta usando la libreria de forma indirecta
)

func main() {

	dns := "juanfervel:password@tcp(localhost:3306)/deb_contact"

	//Abrir una conexion de la base de datos
	db, err := sql.Open("mysql", dns)

	if err != nil {
		log.Fatal(err)
	}

	//Verificar la conexion a la base de datos
	if err := db.Ping(); err != nil {
		log.Fatal(err)
	}
	defer db.Close()

	log.Println("Conexion a la base de datos MySQL exitosa")
}
